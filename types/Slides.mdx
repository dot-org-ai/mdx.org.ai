    {
      title: 'Welcome',
      subtitle: 'Product Launch 2024',
      layout: 'title',
    },
    {
      title: 'Agenda',
      body: [
        'Introduction',
        'New Features',
        'Demo',

      ],
      layout: 'bullets',
    },
    {
      title: 'New Features',

      layout: 'image-right',
    },
  ]}
/>
