## Configuration

```ts
export default {
  name: 'agentic.md',

  // Default model
  model: {
    provider: 'anthropic',
    model: 'claude-sonnet-4-20250514'
  },

  // Tool execution
  tools: {
    timeout: 30000,
    retries: 3
  },

  // Memory
  memory: {
    provider: 'sqlite',
    persist: true
  },

  // Deployment
  deploy: {
    platform: 'cloudflare',
    region: 'auto'
  }
}
```
