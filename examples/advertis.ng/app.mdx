## Configuration

```ts
export default {
  name: 'advertis.ng',

  // AI Models
  models: {
    copywriting: 'claude-sonnet-4-20250514',
    images: 'dall-e-3',
    optimization: 'claude-sonnet-4-20250514'
  },

  // Platforms
  platforms: {
    meta: {
      appId: process.env.META_APP_ID,
      appSecret: process.env.META_APP_SECRET,
      accessToken: process.env.META_ACCESS_TOKEN
    },
    google: {
      credentials: process.env.GOOGLE_ADS_CREDENTIALS,
      developerToken: process.env.GOOGLE_DEVELOPER_TOKEN
    },
    linkedin: {
      clientId: process.env.LINKEDIN_CLIENT_ID,
      clientSecret: process.env.LINKEDIN_CLIENT_SECRET
    }
  },

  // Budgets
  budgets: {
    daily: 100,
    monthly: 3000,
    perExperiment: 500
  }
}
```
