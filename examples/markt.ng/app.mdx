## Configuration

```ts
export default {
  name: 'markt.ng',

  // AI Models
  models: {
    generation: 'claude-sonnet-4-20250514',
    analysis: 'claude-haiku',
    optimization: 'claude-sonnet-4-20250514'
  },

  // Channels
  channels: {
    email: {
      provider: 'sendgrid',
      apiKey: process.env.SENDGRID_API_KEY
    },
    social: {
      meta: process.env.META_ACCESS_TOKEN,
      linkedin: process.env.LINKEDIN_TOKEN
    }
  },

  // Analytics
  analytics: {
    provider: 'mixpanel',
    projectId: process.env.MIXPANEL_PROJECT
  }
}
```
