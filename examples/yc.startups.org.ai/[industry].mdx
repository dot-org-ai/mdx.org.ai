export async function getStaticProps({ params }) {
  const yc = sources.get('https://yc.startups.org.ai')
  const results = await yc.endpoint('search', {}, {
    facetFilters: [[`industries:${params.industry}`]],
    hitsPerPage: 200,
    facets: ['batch', 'status', 'regions']
  })
  return {
    props: {
      industry: params.industry,
      companies: results.hits,
      total: results.nbHits,
      facets: results.facets
    }
  }
}

export async function getStaticPaths() {
  const industries = [
    'B2B', 'Consumer', 'Fintech', 'Healthcare',
    'Developer Tools', 'Education', 'Real Estate',
    'Government', 'Industrials', 'Entertainment',
    'AI/ML', 'Crypto/Web3', 'Climate',
    'Consumer Health', 'Biotech', 'Transportation'
  ]
  return {
    paths: industries.map(industry => ({ params: { industry } })),
    fallback: 'blocking'
  }
}
