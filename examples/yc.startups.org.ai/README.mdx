## API

```ts
import { sources } from '@mdxdb/sources'

const yc = sources.get('https://yc.startups.org.ai')

// Search companies
const results = await yc.endpoint('search', {}, {
  query: 'artificial intelligence',
  hitsPerPage: 20
})

// Browse by batch
const batch = await yc.endpoint('search', {}, {
  filters: 'batch:S23'
})
```
