$type: Batch
$id: https://yc.startups.org.ai/batch/${params.batch}
source:
  endpoint: search
  body:
    query: ''
    filters: batch:${params.batch}
    hitsPerPage: 100
    facets: ['industries', 'status']
  transform: |
    (data) => ({
      $type: 'Batch',
      $id: `https://yc.startups.org.ai/batch/${params.batch}`,
      batch: params.batch,
      companies: data.hits,
      total: data.nbHits,
      industries: data.facets?.industries || {},
      statuses: data.facets?.status || {},
    })
layout: batch
---

export async function getStaticProps({ params }) {
  const yc = sources.get('https://yc.startups.org.ai')
  const results = await yc.endpoint('search', {}, {
    filters: `batch:${params.batch}`,
    hitsPerPage: 200,
    facets: ['industries', 'status']
  })
  return {
    props: {
      batch: params.batch,
      companies: results.hits,
      total: results.nbHits,
      facets: results.facets
    }
  }
}

export async function getStaticPaths() {
  const batches = [

  }
}
