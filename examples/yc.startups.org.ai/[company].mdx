---
$type: Startup
$id: https://yc.startups.org.ai/company/${params.slug}
source:
  endpoint: search
  body:
    filters: slug:${params.slug}
    hitsPerPage: 1
layout: company
---

export function transform(data) {
  const company = data.hits[0]
  if (!company) return null
  return {
    $type: 'Startup',
    $id: `https://yc.startups.org.ai/company/${company.slug}`,
    ...company,
    ycUrl: `https://www.ycombinator.com/companies/${company.slug}`,
  }
}

export async function getStaticProps({ params }) {
  const yc = sources.get('https://yc.startups.org.ai')
  const results = await yc.endpoint('search', {}, {
    filters: `slug:${params.slug}`,
    hitsPerPage: 1
  })
  const company = results.hits[0]
  if (!company) return { notFound: true }
  return { props: { company } }
}

export async function getStaticPaths() {
  const yc = sources.get('https://yc.startups.org.ai')
  const results = await yc.endpoint('browse')
  return {
    paths: results.hits.slice(0, 500).map(c => ({
      params: { slug: c.slug }
    })),
    fallback: 'blocking'
  }
}
