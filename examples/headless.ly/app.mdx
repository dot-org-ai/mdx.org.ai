```mdx
author: alice

tags: [intro, welcome]

```

## Configuration

```ts
export default {
  name: 'headless.ly',

  // Content types
  content: {
    BlogPost: {
      fields: {
        title: { type: 'string', required: true },
        slug: { type: 'string', unique: true },
        author: { type: 'reference', to: 'Author' },
        publishedAt: { type: 'datetime' },
        tags: { type: 'array', of: 'reference', to: 'Tag' },
        body: { type: 'mdx' }
      }
    },
    Author: {
      fields: {
        name: { type: 'string', required: true },
        email: { type: 'string' },
        bio: { type: 'text' },
        avatar: { type: 'image' }
      }
    },
    Tag: {
      fields: {
        name: { type: 'string', required: true },
        slug: { type: 'string', unique: true }
      }
    }
  },

  // AI features
  ai: {
    generation: true,
    seo: true,
    translation: true,
    summarization: true
  },

  // Storage
  storage: {
    provider: 'git',
    repository: 'github:org/content'
  },

  // Publishing
  publishing: {
    preview: true,
    scheduled: true,
    workflows: true
  }
}
```
