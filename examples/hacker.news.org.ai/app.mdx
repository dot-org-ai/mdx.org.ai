export function HNReader() {
  const [feed, setFeed] = useState('personalized')
  const [expanded, setExpanded] = useState({})

  const toggleExpand = (id) => {
    setExpanded(prev => ({ ...prev, [id]: !prev[id] }))
  }

  return (
    <div className="max-w-3xl mx-auto p-6 space-y-6">
      <div className="flex items-center justify-between">
        <h1 className="text-2xl font-bold text-orange-500">HN Enhanced</h1>
        <div className="flex gap-2">
          <Button variant="outline" size="sm">Settings</Button>
          <Button size="sm">Daily Digest</Button>
        </div>
      </div>

      <Tabs value={feed} onValueChange={setFeed}>
        <TabsList>
          <TabsTrigger value="personalized">For You</TabsTrigger>
          <TabsTrigger value="top">Top</TabsTrigger>
          <TabsTrigger value="new">New</TabsTrigger>
          <TabsTrigger value="ask">Ask HN</TabsTrigger>
        </TabsList>

        <TabsContent value={feed} className="mt-4 space-y-4">
          {sampleStories.map(story => (
            <Card key={story.id} className={story.controversial ? 'border-amber-500' : ''}>
              <CardContent className="pt-4">
                <div className="space-y-2">
                  {/* Header */}
                  <div className="flex items-start gap-2">
                    <span className="text-orange-500 font-bold">{story.points}</span>
                    <div className="flex-1">
                      <a href={story.url} className="font-medium hover:text-orange-500">
                        {story.title}
                      </a>
                      <div className="text-sm text-muted-foreground">
                        {story.time} by {story.author} | {story.comments} comments
                      </div>
                    </div>
                  </div>

                  {/* AI Summary */}
                  <div className="bg-muted/50 rounded p-3 text-sm">
                    <div className="flex items-center gap-2 mb-1">
                      <Badge variant="outline" size="sm">AI Summary</Badge>
                      <Badge
                        variant={story.sentiment === 'positive' ? 'success' : story.sentiment === 'mixed' ? 'warning' : 'secondary'}
                        size="sm"
                      >
                        {story.sentiment}
                      </Badge>
                      {story.controversial && (
                        <Badge variant="destructive" size="sm">Controversial</Badge>
                      )}
                    </div>
                    <p>{story.summary}</p>
                  </div>

                  {/* Topics */}
                  <div className="flex gap-2">
                    {story.topics.map(topic => (
                      <Badge key={topic} variant="secondary" size="sm">
                        {topic}
                      </Badge>
                    ))}
                  </div>

                  {/* Thread Summary (expandable) */}
                  <Button
                    variant="ghost"
                    size="sm"
                    onClick={() => toggleExpand(story.id)}
                  >
                    {expanded[story.id] ? 'Hide' : 'Show'} Thread Summary
                  </Button>

                  {expanded[story.id] && (
                    <div className="border-l-2 border-orange-500 pl-4 space-y-2 text-sm">
                      <p className="font-medium">Top Insights from Comments:</p>
                      <ul className="list-disc list-inside space-y-1 text-muted-foreground">
                        <li>Several commenters shared their own implementation experiences</li>
                        <li>Debate about LLVM vs custom backend tradeoffs</li>
                        <li>Resources shared: "Crafting Interpreters", LLVM tutorial</li>
                        <li>Author answered 12 questions about design decisions</li>
                      </ul>
                    </div>
                  )}
                </div>
              </CardContent>
            </Card>
          ))}
        </TabsContent>
      </Tabs>
    </div>
  )
}

<HNReader />

## Features

### AI-Powered Summaries
- **Article TLDR** - Know what it's about before clicking
- **Thread Highlights** - Key insights from 300+ comment threads
- **Sentiment Analysis** - Is the community positive or skeptical?
- **Controversy Detection** - Know when discussions are heated

### Personalized Feed
- Tell us your interests (languages, frameworks, topics)
- AI learns from what you read and engage with
- "More like this" and "Less like this" feedback
- Hide topics you don't care about

### Daily Digest
- Morning email with everything that mattered
- Customizable length and topics
- Weekend edition with best of the week

### Time Savings
- Average HN user: 45 min/day scrolling
- With HN Enhanced: 10 min/day, same or better coverage
- Thread summaries alone save hours on popular posts
