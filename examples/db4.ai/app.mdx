## Configuration

```ts
export default {
  name: 'db4.ai',

  // Database
  database: {
    provider: 'turso',  // or 'postgres', 'sqlite'
    url: process.env.DATABASE_URL
  },

  // Embeddings
  embeddings: {
    provider: 'openai',  // or 'anthropic', 'cohere'
    model: 'text-embedding-3-small',
    dimensions: 1536
  },

  // AI Generation
  generation: {
    provider: 'anthropic',
    model: 'claude-sonnet-4-20250514'
  },

  // Search
  search: {
    mode: 'hybrid',  // 'vector', 'fulltext', 'hybrid'
    vectorWeight: 0.7,
    fulltextWeight: 0.3
  },

  // Get or Generate
  getOrGenerate: {
    enabled: true,
    cache: true,
    ttl: '24h'
  }
}
```
