# Drawer

A bottom sheet that slides up from the bottom of the screen, commonly used on mobile.

## When to Use

- Mobile filters
- Action sheets
- Share menus
- Quick options
- Mobile navigation

## Structure

```html
<aside data-container="drawer" aria-modal="true" aria-hidden="true">
  <article>
    <header>
      <span></span>
      <h2>Filter Results</h2>
    </header>
    <section>
      <!-- Filter options -->
    </section>
    <footer>
      <button>Reset</button>
      <button>Apply Filters</button>
    </footer>
  </article>
</aside>
```

## JavaScript (with Gesture Support)

```js
const drawer = document.querySelector('[data-container="drawer"]')
const article = drawer.querySelector('article')
const handle = drawer.querySelector('header > span')

let startY = 0
let currentY = 0
let isDragging = false

handle.addEventListener('touchstart', (e) => {
  isDragging = true
  startY = e.touches[0].clientY
})

document.addEventListener('touchmove', (e) => {
  if (!isDragging) return
  currentY = e.touches[0].clientY - startY
  if (currentY > 0) {
    article.style.transform = `translateY(${currentY}px)`
  }
})

document.addEventListener('touchend', () => {
  if (!isDragging) return
  isDragging = false

  if (currentY > 100) {
    drawer.setAttribute('aria-hidden', 'true')
  }

  article.style.transform = ''
  currentY = 0
})

// Close on backdrop click
drawer.addEventListener('click', (e) => {
  if (e.target === drawer) {
    drawer.setAttribute('aria-hidden', 'true')
  }
})
```

## Variations

```html
<aside data-container="drawer">
  <article>
    <header>
      <span></span>
    </header>
```
