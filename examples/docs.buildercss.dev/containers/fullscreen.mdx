## When to Use

- Video players
- Photo galleries
- Media presentations
- Immersive reading
- Games
- Interactive demos

## Structure

```html
<aside data-container="fullscreen" aria-hidden="true">
  <main>
    <video src="/media/video.mp4" autoplay controls></video>
  </main>

  <nav data-controls aria-label="Media controls">
    <button data-action="previous">Previous</button>
    <button data-action="play-pause">Play/Pause</button>
    <button data-action="next">Next</button>
  </nav>

  <aside data-overlay>
    <button data-action="settings">Settings</button>
    <button data-action="exit">Exit Fullscreen</button>
  </aside>
</aside>
```

## JavaScript

```js
const fullscreen = document.querySelector('[data-container="fullscreen"]')

// Enter fullscreen
function enterFullscreen(element) {
  if (element.requestFullscreen) {
    element.requestFullscreen()
  } else if (element.webkitRequestFullscreen) {
    element.webkitRequestFullscreen()
  }
}

// Exit fullscreen
function exitFullscreen() {
  if (document.exitFullscreen) {
    document.exitFullscreen()
  } else if (document.webkitExitFullscreen) {
    document.webkitExitFullscreen()
  }
}

// Exit on Escape
document.addEventListener('keydown', (e) => {
  if (e.key === 'Escape') {
    fullscreen.setAttribute('aria-hidden', 'true')
  }
})

// Auto-hide controls
let hideTimeout
fullscreen.addEventListener('mousemove', () => {
  fullscreen.classList.add('show-controls')
  clearTimeout(hideTimeout)
  hideTimeout = setTimeout(() => {
    fullscreen.classList.remove('show-controls')
  }, 3000)
})
```

## Variations

### Photo Gallery

```html
<aside data-container="fullscreen" data-variant="gallery">
  <main>
    <img src="/photo.jpg" alt="Photo" />
  </main>
  <nav data-thumbnails>
    <button data-thumbnail><img src="/thumb1.jpg" alt="" /></button>
    <button data-thumbnail><img src="/thumb2.jpg" alt="" /></button>
  </nav>
</aside>
```

### Presentation Mode

```html
<aside data-container="fullscreen" data-variant="presentation">
  <main>
    <article data-slide="1">
      <h1>Slide Title</h1>
      <p>Slide content...</p>
    </article>
  </main>
  <nav data-slide-nav>
    <span>Slide 1 of 10</span>
  </nav>
</aside>
```
