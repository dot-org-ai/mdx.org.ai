# Fullscreen Container

An edge-to-edge container that takes over the entire viewport for immersive experiences.

<Preview>
  <div data-container="fullscreen">
    <aside>
      <button>⚙️ Settings</button>
      <button>✕ Exit</button>
    </aside>
    <main>
      <figure>▶</figure>
      <h2>Now Playing</h2>
      <p>Beautiful Sunset - Nature Sounds</p>
    </main>
    <footer>
      <progress value="35" max="100"></progress>
      <nav aria-label="Media controls">
        <button>⏮</button>
        <button aria-current="true">⏸</button>
        <button>⏭</button>
      </nav>
    </footer>
  </div>
</Preview>

## When to Use

- Video players
- Photo galleries
- Media presentations
- Immersive reading
- Games
- Interactive demos

## Structure

```html
<aside data-container="fullscreen" aria-hidden="true">
  <aside>
    <button>Settings</button>
    <button>Exit Fullscreen</button>
  </aside>

  <main>
    <video src="/media/video.mp4" autoplay controls></video>
  </main>

  <nav aria-label="Media controls">
    <button>Previous</button>
    <button>Play/Pause</button>
    <button>Next</button>
  </nav>
</aside>
```

## JavaScript

```js
const fullscreen = document.querySelector('[data-container="fullscreen"]')

// Enter fullscreen
function enterFullscreen(element) {
  if (element.requestFullscreen) {
    element.requestFullscreen()
  } else if (element.webkitRequestFullscreen) {
    element.webkitRequestFullscreen()
  }
}

// Exit fullscreen
function exitFullscreen() {
  if (document.exitFullscreen) {
    document.exitFullscreen()
  } else if (document.webkitExitFullscreen) {
    document.webkitExitFullscreen()
  }
}

// Exit on Escape
document.addEventListener('keydown', (e) => {
  if (e.key === 'Escape') {
    fullscreen.setAttribute('aria-hidden', 'true')
  }
})

// Auto-hide controls
let hideTimeout
fullscreen.addEventListener('mousemove', () => {
  fullscreen.classList.add('show-controls')
  clearTimeout(hideTimeout)
  hideTimeout = setTimeout(() => {
    fullscreen.classList.remove('show-controls')
  }, 3000)
})
```

## Variations

### Photo Gallery

```html
<aside data-container="fullscreen">
  <main>
    <img src="/photo.jpg" alt="Photo" />
  </main>
  <nav>
    <button><img src="/thumb1.jpg" alt="" /></button>
    <button><img src="/thumb2.jpg" alt="" /></button>
  </nav>
</aside>
```

### Presentation Mode

```html
<aside data-container="fullscreen">
  <main>
    <article>
      <h1>Slide Title</h1>
      <p>Slide content...</p>
    </article>
  </main>
  <nav>
    <span>Slide 1 of 10</span>
  </nav>
</aside>
```
