## JavaScript

```js
// Toggle switches
document.querySelectorAll('[data-view="settings"] button[role="switch"]').forEach(btn => {
  btn.addEventListener('click', () => {
    const checked = btn.getAttribute('aria-checked') === 'true'
    btn.setAttribute('aria-checked', !checked)
  })
})

// Tab navigation
document.querySelectorAll('[data-view="settings"] > nav button').forEach(btn => {
  btn.addEventListener('click', () => {
    document.querySelectorAll('[data-view="settings"] > nav button').forEach(b => {
      b.removeAttribute('aria-selected')
    })
    btn.setAttribute('aria-selected', 'true')
  })
})
```
