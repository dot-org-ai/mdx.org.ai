## JavaScript

```js
const calendar = document.querySelector('[data-view="calendar"]')
const monthYear = calendar.querySelector('header h1')
const prevBtn = calendar.querySelector('button[aria-label="Previous month"]')
const nextBtn = calendar.querySelector('button[aria-label="Next month"]')

let currentDate = new Date()

function renderCalendar() {
  const month = currentDate.toLocaleString('default', { month: 'long' })
  const year = currentDate.getFullYear()
  monthYear.textContent = `${month} ${year}`
  // Update table cells...
}

prevBtn.addEventListener('click', () => {
  currentDate.setMonth(currentDate.getMonth() - 1)
  renderCalendar()
})

nextBtn.addEventListener('click', () => {
  currentDate.setMonth(currentDate.getMonth() + 1)
  renderCalendar()
})
```
