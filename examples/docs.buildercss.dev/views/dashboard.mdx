```html

  <header>
    <h1>Dashboard</h1>
    <menu>
      <button>Export</button>
      <button>Settings</button>
    </menu>
  </header>

  <section aria-label="Metrics">
    <article data-metric>
      <header><span>Total Users</span></header>
      <data value="12345">12,345</data>
      <footer>+12% from last month</footer>
    </article>
    <!-- More metrics -->
  </section>

  <section aria-label="Charts">
    <article>
      <header><h3>Revenue</h3></header>
      <figure><!-- Chart component --></figure>
    </article>
  </section>

  <section aria-label="Recent activity">
    <article>
      <header><h3>Recent Activity</h3></header>
      <ul data-activity-list><!-- Activity items --></ul>
    </article>
  </section>
</article>
```
