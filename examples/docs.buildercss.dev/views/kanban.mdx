## JavaScript

```js
const cards = document.querySelectorAll('[data-view="kanban"] article')
const columns = document.querySelectorAll('[data-view="kanban"] section')

cards.forEach(card => {
  card.addEventListener('dragstart', () => {
    card.setAttribute('aria-grabbed', 'true')
  })

  card.addEventListener('dragend', () => {
    card.removeAttribute('aria-grabbed')
  })
})

columns.forEach(column => {
  column.addEventListener('dragover', (e) => {
    e.preventDefault()
    const dragging = document.querySelector('[aria-grabbed="true"]')
    column.appendChild(dragging)
  })
})
```
