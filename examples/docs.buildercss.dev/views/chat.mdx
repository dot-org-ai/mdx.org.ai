## JavaScript

```js
const chat = document.querySelector('[data-view="chat"]')
const section = chat.querySelector('section')
const input = chat.querySelector('footer input')
const sendBtn = chat.querySelector('footer button:last-child')

function addMessage(text, isSelf = false) {
  const div = document.createElement('div')
  if (isSelf) div.dataset.self = 'true'

  const p = document.createElement('p')
  p.textContent = text

  const time = document.createElement('time')
  time.textContent = new Date().toLocaleTimeString([], { hour: '2-digit', minute: '2-digit' })

  div.appendChild(p)
  div.appendChild(time)
  section.appendChild(div)
  section.scrollTop = section.scrollHeight
}

sendBtn.addEventListener('click', () => {
  if (input.value.trim()) {
    addMessage(input.value, true)
    input.value = ''
  }
})

input.addEventListener('keypress', (e) => {
  if (e.key === 'Enter') sendBtn.click()
})
```
