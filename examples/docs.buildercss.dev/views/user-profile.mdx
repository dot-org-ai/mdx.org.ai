## Structure

```html
<article data-view="user-profile">
  <header>
    <img src="/avatar.jpg" alt="" data-avatar />
    <button data-action="change-avatar">Change</button>
    <h1>John Doe</h1>
    <p>john@example.com</p>
  </header>

  <section data-section="personal">
    <h2>Personal Information</h2>
    <form>
      <div data-field-group>
        <div data-field>
          <label for="firstName">First Name</label>
          <input type="text" id="firstName" value="John" />
        </div>
        <div data-field>
          <label for="lastName">Last Name</label>
          <input type="text" id="lastName" value="Doe" />
        </div>
      </div>
      <div data-field>
        <label for="email">Email</label>
        <input type="email" id="email" value="john@example.com" />
      </div>
      <div data-field>
        <label for="bio">Bio</label>
        <textarea id="bio" rows="3"></textarea>
      </div>
    </form>
  </section>

  <section data-section="password">
    <h2>Change Password</h2>
    <form>
      <div data-field>
        <label for="currentPassword">Current Password</label>
        <input type="password" id="currentPassword" />
      </div>
      <div data-field>
        <label for="newPassword">New Password</label>
        <input type="password" id="newPassword" />
      </div>
    </form>
  </section>

  <footer>
    <button data-button="destructive">Delete Account</button>
    <button data-button="primary">Save Changes</button>
  </footer>
</article>
```
