```html
  <header>
    <h1>Projects</h1>
    <div data-actions>
      <input type="search" placeholder="Search..." />
      <select>
        <option>All Status</option>
        <option>Active</option>
        <option>Archived</option>
      </select>
      <button data-button="primary">New Project</button>
    </div>
  </header>

  <section data-content>
    <ul data-list>
      <li data-list-item>
        <div data-list-item-content>
          <h3>Project Alpha</h3>
          <p>Description of the project</p>
          <div data-meta>
            <span data-badge="success">Active</span>
            <span>Updated 2 hours ago</span>
          </div>
        </div>
        <div data-list-item-actions>
          <button data-button="ghost">Edit</button>
          <button data-button="ghost">Delete</button>
        </div>

    </ul>
  </section>

  <footer>
    <nav data-pagination>
      <button disabled>Previous</button>
      <span>Page 1 of 5</span>
      <button>Next</button>
    </nav>
  </footer>
</article>
```
