## JavaScript

```js
document.querySelectorAll('[data-section="faq"] dt button').forEach(button => {
  button.addEventListener('click', () => {
    const expanded = button.getAttribute('aria-expanded') === 'true'
    button.setAttribute('aria-expanded', !expanded)
    const dd = button.closest('div').querySelector('dd')
    dd.hidden = expanded
  })
})
```
