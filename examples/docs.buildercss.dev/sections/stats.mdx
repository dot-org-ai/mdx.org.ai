## When to Use

- Impact showcases
- Product statistics
- Company milestones
- Performance metrics

## Variations

### With Icons

```html
<section data-section="stats">
  <article>
    <figure>
      <svg><!-- Users icon --></svg>
    </figure>
    <data value="10000">10K+</data>
    <p>Active Users</p>
  </article>
</section>
```

### Animated Counters

```html
<section data-section="stats">
  <article>
    <data value="10000">0</data>
    <p>Active Users</p>
  </article>
</section>
```

```js
// Counter animation
const counters = document.querySelectorAll('[data-section="stats"] data')
const observer = new IntersectionObserver(entries => {
  entries.forEach(entry => {
    if (entry.isIntersecting) {
      const counter = entry.target
      const target = parseInt(counter.getAttribute('value'))
      animateCounter(counter, target)
    }
  })
})

counters.forEach(counter => observer.observe(counter))
```

### With Background

```html
<section data-section="stats" aria-current="true">
  <div>
    <!-- Stats with light text -->
  </div>
</section>
```

```css

  background: var(--foreground);

}

[data-section="stats"][aria-current] article data {
  color: var(--background);
}
```
