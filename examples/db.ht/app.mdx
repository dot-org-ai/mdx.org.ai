## Configuration

```ts
export default {
  name: 'db.ht',

  // Database connection
  database: {
    provider: 'sqlite',  // or 'postgres', 'turso'
    url: process.env.DATABASE_URL
  },

  // URL generation
  baseUrl: 'https://db.ht',

  // ID format
  id: {
    prefix: true,      // u_abc123 vs abc123
    format: 'nanoid'   // or 'uuid', 'cuid'
  },

  // Response format
  format: {
    id: '$id',
    ref: '$ref',
    type: '$type'
  },

  // Pagination
  pagination: {
    defaultLimit: 20,
    maxLimit: 100
  }
}
```
