## API

```ts
import { sources } from '@mdxdb/sources'

const schema = sources.get('https://schema.org.ai')

// Get type definition
const person = await schema.endpoint('type', { type: 'Person' })

// Get all types as CSV
const types = await schema.endpoint('types')

// Get full vocabulary
const vocab = await schema.endpoint('vocabulary')
```

## Convert to MDXLD

```ts
import { jsonldToMDXLD } from '@mdxdb/sources/jsonld'

const jsonld = {
  "@context": "https://schema.org",
  "@type": "Person",
  "name": "John Doe"
}

const mdxld = jsonldToMDXLD(jsonld)
// { $type: 'Person', $id: '...', name: 'John Doe' }
```
