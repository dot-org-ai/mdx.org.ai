## Configuration

```ts
export default {
  name: 'builder.domains',

  // Free tier limits
  free: {
    maxDomains: 3,
    allowedSuffixes: ['.builder.domains']
  },

  // Pro features
  pro: {
    customDNS: true,
    ssl: 'automatic',
    analytics: true
  },

  // Registrar integrations
  registrars: {
    cloudflare: true,
    namecheap: true,
    godaddy: true
  }
}
```
