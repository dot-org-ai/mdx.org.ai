{cta.subheadline}

```tsx
const CTAForm = () => {
  'use client'
  const [email, setEmail] = useState('')
  const [loading, setLoading] = useState(false)
  const [submitted, setSubmitted] = useState(false)

  const submit = async () => {
    setLoading(true)

    await db.Leads.create({
      $type: 'Lead',
      email,
      source: slug,
      createdAt: new Date()
    })

    track('landing.signup', { page: slug, email })

    await send('lead.created', { email, source: slug })

    setLoading(false)
    setSubmitted(true)
  }

  if (submitted) {
    return <p>Thanks! We'll be in touch soon.</p>
  }

  return (
    <form onSubmit={e => { e.preventDefault(); submit() }}>
      <input
        type="email"
        value={email}
        onChange={e => setEmail(e.target.value)}
        placeholder="Enter your email"
        required
      />
      <button disabled={loading}>
        {loading ? 'Submitting...' : cta.buttonLabel}
      </button>
    </form>
  )
}
```
