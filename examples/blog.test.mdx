# Tests for Getting Started Blog Post

This is a **companion test file** for `blog.mdx`.
It tests the utility functions defined in the blog post.

## String Utility Tests

### Capitalize Tests

```ts test name="capitalize capitalizes first letter"
function capitalize(str: string): string {
  if (!str) return ''
  return str.charAt(0).toUpperCase() + str.slice(1)
}

expect(capitalize('hello')).toBe('Hello')
expect(capitalize('world')).toBe('World')
expect(capitalize('a')).toBe('A')
```

```ts test name="capitalize handles edge cases"
function capitalize(str: string): string {

  return str.charAt(0).toUpperCase() + str.slice(1)
}

expect(capitalize('')).toBe('')

```

### Slugify Tests

```ts test name="slugify creates URL-friendly slugs"
function slugify(text: string): string {
  return text
    .toLowerCase()
    .trim()
    .replace(/[^\w\s-]/g, '')
    .replace(/[\s_-]+/g, '-')
    .replace(/^-+|-+$/g, '')
}

expect(slugify('Hello World')).toBe('hello-world')
expect(slugify('Getting Started with MDX')).toBe('getting-started-with-mdx')
```


```ts test name="slugify normalizes dashes"
function slugify(text: string): string {
```
