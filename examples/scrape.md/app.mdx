## Configuration

```ts
export default {
  name: 'scrape.md',

  // Output format
  format: 'markdown',  // or 'text', 'html'

  // Content extraction
  extraction: {
    removeNav: true,
    removeFooter: true,
    removeAds: true,
    extractMain: true
  },

  // JavaScript rendering
  javascript: {
    enabled: true,
    waitFor: 'networkidle',
    timeout: 10000
  },

  // Caching
  cache: {
    enabled: true,
    ttl: '1h'
  },

  // Rate limiting
  rateLimit: {
    requests: 100,
    window: '1m'
  }
}
```
