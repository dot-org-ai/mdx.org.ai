## Configuration

```ts
export default {
  name: 'api.ht',

  // Response format
  format: {
    id: '$id',
    ref: '$ref',
    type: '$type'
  },

  // Link generation
  links: {
    self: (resource) => resource.$id,
    collection: (type) => `/${type}`,
    related: (resource, rel) => `${resource.$id}/${rel}`
  },

  // Content negotiation
  accept: ['application/json', 'text/html', 'text/markdown'],

  // CORS
  cors: {
    origins: ['*'],
    methods: ['GET', 'POST', 'PUT', 'PATCH', 'DELETE']
  }
}
```
