## Configuration

```ts
export default {
  name: 'workflow.md',

  // Execution
  execution: {
    timeout: '5m',
    retries: 3,
    backoff: 'exponential'
  },

  // Triggers
  triggers: {
    webhook: true,
    schedule: true,
    events: true
  },

  // Integrations
  integrations: {
    slack: process.env.SLACK_TOKEN,
    github: process.env.GITHUB_TOKEN,
    stripe: process.env.STRIPE_KEY
  },

  // Storage
  storage: {
    provider: 'sqlite',
    path: './workflows.db'
  }
}
```
